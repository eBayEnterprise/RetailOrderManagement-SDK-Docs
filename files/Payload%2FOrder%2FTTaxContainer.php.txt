namespace eBayEnterprise\RetailOrderManagement\Payload\Order;

trait TTaxContainer
{
    /** @var ITaxIterable */
    protected $taxes;
    /** @var string */
    protected $taxClass;

    public function getTaxes()
    {
        return $this->taxes;
    }

    public function setTaxes(ITaxIterable $taxes)
    {
        $this->taxes = $taxes;
        return $this;
    }

    public function getTaxClass()
    {
        return $this->taxClass;
    }

    public function setTaxClass($taxClass)
    {
        $this->taxClass = $taxClass;
        return $this;
    }

    /**
     * Serialize the tax class and taxes in the container.
     *
     * @return string
     */
    protected function serializeTaxData()
    {
        $taxClass = $this->getTaxClass();
        $taxes = $this->getTaxes();
        return (!is_null($taxClass) || count($taxes))
            ? '<TaxData>'
                . $this->serializeOptionalValue('TaxClass', $this->getTaxClass())
                . $this->getTaxes()->serialize()
                . '</TaxData>'
            : '';
    }
}
